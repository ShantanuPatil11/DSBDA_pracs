def handle_out(column):
    Q1=column.quantile(0.25)
    Q2=column.quantile(0.75)
    IQR=Q2-Q1
    lower=Q1-1.5*IQR
    upper=Q2+1.5*IQR
    return np.where((column<lower)|(column>upper),column.median(),column)

from sklearn.feature_extraction.text import TfidfVectorizer

# Sample text
text = "Virat Kohli is one of the greatest batsmen in cricket"

# Tokenization
tokens = text.split()
print("Tokens:", tokens)

# Stopword Removal
stop = ['is', 'one', 'of', 'the', 'in']
words = [i for i in tokens if i.lower() not in stop]
print("\nAfter Stopword Removal:", words)

# TF-IDF
tfidf = TfidfVectorizer()
x = tfidf.fit_transform([text])

# Sample Logs
logs = [
    "ERROR",
    "INFO",
    "ERROR",
    "WARNING",
    "INFO",
    "ERROR"
]

# MAP
mapped = [(log, 1) for log in logs]
print("Mapped Data:")
print(mapped)

# REDUCE
result = {}

for key, value in mapped:
    result[key] = result.get(key, 0) + value

print("\nReduced Data:")
print(result)
